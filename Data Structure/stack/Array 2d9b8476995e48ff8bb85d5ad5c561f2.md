# Array

create함수:스택 생성

IsEmpty함수:스택이 비었는지 확인

IsFull함수:스택이 가득 찼는지 확인

display함수:스택의 top부터 순서대로 출력

push함수:스택에 값추가

pop함수:스택에서 값을 빼서 반환

peek함수:index에 해당하는 값 반환, index가 1이면, 스택의 맨위의 값을 의미한다.

StackTop함수:스택의 맨위값을 반환

create, push, pop은 스택의 값을 변경하기 때문에 매개변수는 포인터를 사용하였다.

나머지 함수는 값을 반환하거나, 출력하는 역할이기 때문에 매개변수는 변수를 사용하였다.

```c
#include <stdio.h>
#include <stdlib.h>

struct Stack
{
    int size;
    int top;
    int *S;
};

void create(struct Stack *st)
{
    printf("Enter size of stack :");
    scanf("%d", &st->size);
    
    st->S=(int *)malloc(st->size*sizeof(int *));
    st->top=-1;
}

int IsEmpty(struct Stack st)
{
    if(st.top==-1)
        return 1;
    else
        return 0;
}

int IsFull(struct Stack st)
{
    if(st.top==st.size-1)
        return 1;
    else
        return 0;
}

void display(struct Stack st)
{
    for(int i=st.top;i>=0;i--)
    {
        printf("%d ", st.S[i]);
    }
    printf("\n");
}

void push(struct Stack *st, int data)
{
    if(IsFull(*st))
    {
        printf("stack overflow\n");
        return;
    }
    st->top++;
    st->S[st->top]=data;
}

int pop(struct Stack *st)
{
    int x=-1;
    if(IsEmpty(*st))
    {
        printf("stack underflow\n");
        return -1;
    }
    x=st->S[st->top--];
    return x;
}

int peek(struct Stack st, int index)
{
    int x=-1;
    if(st.top-index+1<0)
    {
        printf("index is invalid\n");
        return -1;
    }
    else
        x=st.S[st.top-index+1];
    return x;
}

int StackTop(struct Stack st)
{
    if(IsEmpty(st))
        return -1;
    else
        return st.S[st.top];
}

int main()
{
    struct Stack st;
    
    create(&st);
    push(&st, 1);
    push(&st, 2);
    push(&st, 3);
    push(&st, 4);
    push(&st, 5);
    printf("pop :%d\n", pop(&st));
    printf("StackTop :%d\n", StackTop(st));
    printf("peek :%d\n", peek(st, 2));
    display(st);

    return 0;
}

```
