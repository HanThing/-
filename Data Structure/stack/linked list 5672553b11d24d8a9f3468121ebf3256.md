# linked list

push함수:스택에 데이터 삽입

pop함수:스택의 맨위 데이터 삭제 및 반환

display함수:출력

```c
#include <stdio.h>
#include <stdlib.h>

struct Node
{
    int data;
    struct Node *next;
}*top=NULL;

void push(int data)
{
    struct Node *t;
    t=(struct Node *)malloc(sizeof(struct Node));
    
    if(t==NULL)
        printf("stack is full\n");
    else
    {
        t->data=data;
        t->next=top;
        top=t;
    }
    
}

int pop()
{
    int x=-1;
    struct Node *t;
    
    if(top==NULL)
        printf("Stack is Empty\n");
    else
    {
        t=top;
        top=t->next;
        x=t->data;
        free(t);
    }
    return x;
}

void display()
{
    struct Node *p;
    p=top;
    while(p)
    {
        printf("%d ", p->data);
        p=p->next;
    }
    printf("\n");
}

int main()
{
    push(1);
    push(2);
    push(3);
    push(4);
    push(5);
    
    display();
    
    printf("%d\n", pop());
    
    display();
    return 0;
}
```
