# 병합(Merging)

merge함수:정렬된 연결리스트 2개를 정렬되도록 합치는 함수

주어진 연결리스트는 first와 second로 접근할 수 있고, 합쳐진 연결리스트는 third로 접근할 수 있다.

```c
#include <stdio.h>
#include <stdlib.h>

struct NODE
{
    int data;
    struct NODE *next;
}*first=NULL, *second=NULL, *third=NULL;

void CreateFirst(int A[], int n)
{
    struct NODE *t, *last;
    first=(struct NODE *)malloc(sizeof(struct NODE));
    first->data=A[0];
    first->next=NULL;
    last=first;
    
    for(int i=1; i<n ; i++)
    {
        t=(struct NODE *)malloc(sizeof(struct NODE));
        t->data=A[i];
        t->next=NULL;
        last->next=t;
        last=t;
    }
    
}

void CreateSecond(int A[], int n)
{
    struct NODE *t, *last;
    second=(struct NODE *)malloc(sizeof(struct NODE));
    second->data=A[0];
    second->next=NULL;
    last=second;
    
    for(int i=1; i<n ; i++)
    {
        t=(struct NODE *)malloc(sizeof(struct NODE));
        t->data=A[i];
        t->next=NULL;
        last->next=t;
        last=t;
    }
    
}

void display(struct NODE *p)
{
    if(p!=NULL)
    {
        printf("%d ", p->data);
        display(p->next);
    }
}

void merge(struct NODE *p, struct NODE *q)
{
    struct NODE *last;
    if(p->data < q->data)
    {
        third=last=p;
        p=p->next;
        third->next=NULL;
    }
    else
    {
        third=last=q;
        q=q->next;
        third->next=NULL;
    }
    
    while(p!=NULL&&q!=NULL)
    {
        if(p->data < q->data)
        {
            last->next=p;
            last=p;
            p=p->next;
            last->next=NULL;
        }
        else
        {
            last->next=q;
            last=q;
            q=q->next;
            last->next=NULL;
        }
    }
    if(p!=NULL)
        last->next=p;
    if(q!=NULL)
        last->next=q;
}

int main()
{
    int A[]={10, 30, 50, 70, 90};
    int B[]={20, 40, 60, 80, 100};
    CreateFirst(A, 5);
    display(first);
    
    printf("\n");
    
    CreateSecond(B, 5);
    display(second);
    
    printf("\n");
    
    merge(first, second);
    
    display(third);

    return 0;
}
```
