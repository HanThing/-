# 원하는 노드 찾기

key값에 해당하는 노드를 찾고, 그노드를 first 노드로 만든다.

생성:create함수

노드를 찾고, 맨앞으로보내기:search함수

출력:display함수

```c
#include<stdio.h>
#include<stdlib.h>

struct NODE 
{
    int data;
    struct NODE *next;
}*first=NULL;

void create(int A[], int n)
{
    struct NODE *t, *last;
    first = (struct NODE *)malloc(sizeof(struct NODE));
    first->data = A[0];
    first->next = NULL;
    last = first;
    
    for(int i=1 ; i<n ; i++)
    {
        t = (struct NODE *)malloc(sizeof(struct NODE));
        t->data = A[i];
        t->next = NULL;
        last->next = t;
        last = t;
    }
}

struct NODE* search(struct NODE *p, int key)
{
    struct NODE *q=NULL;
    
    while(p!=NULL)
    {
        if(p->data==key)
        {
            q->next = p->next;
            p->next = first;
            first = p;
            return p;
        }
        
        q = p;
        p = p->next;
    }
    return NULL;
}

void display(struct NODE *p, int n)
{
    while(p!=NULL)
    {
        printf("%d ", p->data);
        p = p->next;
    }
}

int main()
{
    int A[] = {1, 2, 3, 4, 5};
    create(A, 5);
    display(first, 5);
    search(first, 4);
    display(first, 5);
    

    return 0;
}
```
