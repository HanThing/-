# 정렬된 연결리스트에 삽입

SortedInsert함수:정렬이 유지되도록 삽입

count함수:노드개수세기

display함수:출력

```c
#include<stdio.h>
#include<stdlib.h>

struct NODE 
{
    int data;
    struct NODE *next;
}*first=NULL;

void SortedInsert(struct NODE *p, int Data)
{
    struct NODE *t, *q=NULL;
    t = (struct NODE *)malloc(sizeof(struct NODE));
    t->data = Data;
    t->next = NULL;
    
    if(first==NULL)
        first = t;
    else
    {
        while(p!=NULL && p->data<Data)
        {
            q=p;
            p = p->next;
        }
        if(first->data>Data)
        {
            t->next = p;
            first = t;
            return;
        }
        t->next = p;
        q->next = t;
    }
}

int count(struct NODE*p)
{
    int count=0;
    while(p!=NULL)
    {
        count++;
        p = p->next;
    }
    return count;
}

void display(struct NODE *p, int n)
{
    while(p!=NULL)
    {
        printf("%d ", p->data);
        p = p->next;
    }
}

int main()
{
    
    SortedInsert(first, 15);
    SortedInsert(first, 9);
    SortedInsert(first, 10);
    SortedInsert(first, 16);
    display(first, count(first));
    
    return 0;
}
```
